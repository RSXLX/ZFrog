  // çŠ¶æ€
  address: string | undefined;
  shortAddress: string;

  isConnecting: boolean;

export function useWallet(): UseWalletReturn {

  const { address, isConnected, isConnecting, connector } = useAccount();
  const { disconnect: wagmiDisconnect } = useDisconnect();
  const chainId = useChainId();
  const { switchChain } = useSwitchChain();
  
  const { 
    session, 
    saveSession, 
    clearSession, 
    isSessionValid,
    updateLastActive 
  } = useSessionStore();

        connector: connector.id,
        connectedAt: Date.now(),
        lastActiveAt: Date.now(),
        metadata: {
          name: connector.name,
          icon: connector.icon,
        },
      });

  }, [isConnected, address, connector, chainId, saveSession]);
  
  // å®šæœŸæ›´æ–°æ´»è·ƒæ—¶é—´
  useEffect(() => {
    if (!isConnected) return;
    
    const interval = setInterval(() => {
      updateLastActive();

    return () => clearInterval(interval);
  }, [isConnected, updateLastActive]);
  
  // è¿æ¥

    open();
  }, [open]);
  
  // æ–­å¼€è¿æ¥
  const disconnect = useCallback(() => {
    wagmiDisconnect();
    clearSession();
    close();
  }, [wagmiDisconnect, clearSession, close]);
  
  // åˆ‡æ¢åˆ° ZetaChain
  const switchToZetaChain = useCallback(async () => {
    if (switchChain) {
      await switchChain({ chainId: zetachainAthens.id });
    }
  }, [switchChain]);
  
  return {
    // çŠ¶æ€
    address,
    shortAddress: address ? formatAddress(address) : '',

    address,
    abi,
    functionName,
    args = [],
    value,
  }: {
    address: `0x${string}`;
    abi: Abi;
    functionName: string;
    args?: any[];
    value?: bigint;
  }) => {
    try {
      setState({ status: 'pending' });
      
      const hash = await writeContractAsync({
        address,
        abi,
        functionName,
        args,
        value,
      });
      
      setState({ status: 'confirming', hash });
      console.log('ğŸ“¤ Transaction sent:', hash);
      
      return hash;
    } catch (error: any) {
      console.error('âŒ Transaction failed:', error);
      setState({ status: 'error', error });
      options.onError?.(error);
      throw error;
    }
  }, [writeContractAsync, options]);
  
  // ç›‘å¬äº¤æ˜“ç¡®è®¤
  useEffect(() => {
    if (receipt && state.status === 'confirming') {
      setState(prev => ({ ...prev, status: 'success', receipt }));

      options.onSuccess?.(state.hash!, receipt);
    }
  }, [receipt, state.status, state.hash, options]);
  
  // é‡ç½®çŠ¶æ€
  const reset = useCallback(() => {
    setState({ status: 'idle' });
  }, []);
  
  return {

    isLoading: state.status === 'pending' || state.status === 'confirming',
    execute,
    reset,
  };
}
